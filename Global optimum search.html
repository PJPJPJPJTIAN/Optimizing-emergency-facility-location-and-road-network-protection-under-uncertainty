def subproblem(OD, T, orgin, destination, Nodes, cases, case, x_s, y_s,
               short_paths):
    model = Model("subproblem")
    g = {}
    for od in OD.keys():
        for t in T.keys():
            index = 'g' + '_' + od[0] + ',' + od[1] + '_' + t[0] + ',' + t[1]
            g[od, t] = model.addVar(lb=0,
                                    ub=1,
                                    vtype=GRB.CONTINUOUS,
                                    name=index)

    for od in OD.keys():
        # constraint1 1 and constraint 2
        lhs_1 = LinExpr(0)
        lhs_2 = LinExpr(0)
        for t in T.keys():
            if (t[0] == od[0]):
                lhs_1.addTerms(1, g[od, t])
            if (t[1] == od[1]):
                lhs_2.addTerms(1, g[od, t])
        model.addConstr(lhs_1 == 1)
        model.addConstr(lhs_2 == 1)

        # constraints 3
        for node in Nodes:
            lhs = LinExpr(0)
            if (node != od[0] and node != od[1]):
                for t in T.keys():
                    if (t[1] == node):
                        lhs.addTerms(1, g[od, t])
                    if (t[0] == node):
                        lhs.addTerms(-1, g[od, t])
                model.addConstr(lhs == 0)

    time = {}
    for od in OD.keys():
        index = 'time' + '_' + od[0] + ',' + od[1]
        time[od] = model.addVar(lb=0, ub=40, vtype=GRB.CONTINUOUS, name=index)

    for od in OD.keys():
        model.addConstr(short_paths[od] >= time[od])

    for od in OD.keys():
        lhs = 0
        for t in T.keys():
            lhs += (T[t] + cases[case][t]) * g[od, t] - cases[case][t] * y_s[t] * g[od, t]
        model.addConstr(lhs <= time[od], name='y_' + od[0] + '_' + od[1])

    z = {}
    for od in OD.keys():
        index = 'z' + '_' + od[0] + ',' + od[1]
        z[od[0], od[1]] = model.addVar(lb=0,
                                       ub=1,
                                       vtype=GRB.CONTINUOUS,
                                       name=index)

    for i in orgin:
        lhs = LinExpr(0)
        for j in destination:
            lhs.addTerms(1, z[i, j])
        model.addConstr(lhs == 1)

    for i in orgin:
        for j in destination:
            model.addConstr(z[i, j] <= x_s[j],
                            name='x_' + str(i) + '_' + str(j))

    objective = 0
    for od in OD.keys():
        objective += time[od] * z[od[0], od[1]]

    model.setObjective(objective, GRB.MINIMIZE)
    model.write("Master1.lp")
    return model

def short_path(T, OD, y_s, cases, case):
    short_paths = {}
    for od in OD.keys():
        mm = Model('short_path')
        gg = {}
        for t in T.keys():
            index = 'gg' + t[0] + ',' + t[1]
            gg[t] = mm.addVar(lb=0, ub=1, vtype=GRB.CONTINUOUS, name=index)

        # constraint1 1 and constraint 2
        lhs_1 = LinExpr(0)
        lhs_2 = LinExpr(0)
        for t in T.keys():
            if (t[0] == od[0]):
                lhs_1.addTerms(1, gg[t])
            if (t[1] == od[1]):
                lhs_2.addTerms(1, gg[t])
        mm.addConstr(lhs_1 == 1)
        mm.addConstr(lhs_2 == 1)

        # constraints 3
        for node in Nodes:
            lhs = LinExpr(0)
            if (node != od[0] and node != od[1]):
                for t in T.keys():
                    if (t[1] == node):
                        lhs.addTerms(1, gg[t])
                    if (t[0] == node):
                        lhs.addTerms(-1, gg[t])
                mm.addConstr(lhs == 0)

        ob = LinExpr(0)
        for t in T.keys():
            ob.addTerms(T[t] + cases[case][t] - y_s[t], gg[t])

        mm.setObjective(ob, GRB.MINIMIZE)
        mm.Params.OutputFlag = 0
        mm.optimize()
        short_paths[od] = mm.objVal
    return short_paths

def dual_variables(OD, T, cases, case, duals, variable, x, y):
    dual_equality = 0
    for od in OD.keys():
        dual_single = 0
        index_time = 'time' + '_' + od[0] + ',' + od[1]
        name = 'y_' + od[0] + '_' + od[1]
        for t in T.keys():
            index_g = 'g' + '_' + od[0] + ',' + od[1] + '_' + t[0] + ',' + t[1]
            dual_single += (T[t] + cases[case][t]) * variable[index_g] - cases[
                case][t] * variable[index_g] * y[t]
        dual_single = max(duals[name],
                          -duals[name]) * (dual_single - variable[index_time])
        dual_equality += dual_single
    for i in orgin:
        for j in destination:
            index_z = 'z' + '_' + i + ',' + j
            name = 'x_' + str(i) + '_' + str(j)
            dual_equality += max(duals[name],
                                 -duals[name]) * (variable[index_z] - x[j])

    return dual_equality
